38/*
39 * Summary:
40 *
41 * Function Parameter Type Parameters Description
42 *
43 * SR Set Dominant Bistable
44 * S1 BOOL Input Sets the output Q1.
45 * R BOOL Input Resets the output Q1.
46 * Q1 BOOL Output The output value.
47 *
48 * RS Reset Dominant Bistable
49 * S BOOL Input Sets the output Q1.
50 * R1 BOOL Input Resets the output Q1.
51 * Q1 BOOL Output The output value.
52 *
53 * R_TRIG Rising Edge Detector
54 * CLK BOOL Input A rising edge sets the output Q for one execution.
55 * Q BOOL Output The output value.
56 *
57 * F_TRIG Falling Edge Detector
58 * CLK BOOL Input A falling edge sets the output Q for one execution.
59 * Q BOOL Output The output value.
60 *
61 * CTU Counts up from zero to a preset value.
62 * CU BOOL R_EDGE Input A rising edge increments the counter value CV by 1.
63 * R BOOL Input Resets the output Q and the counter value CV.
64 * PV INT Input The counter limit value.
65 * Q BOOL Output Set to true when the counter value CV reaches PV.
66 * CV INT Input The current counter value.
67 *
68 * CTD Counts down from a preset value to zero.
69 * CD BOOL R_EDGE Input A rising edge decrements the counter value CV by 1.
70 * LD BOOL Input Resets the output Q and the counter value CV.
71 * PV INT Input The counter limit value.
72 * Q BOOL Output Set to true when the counter value CV reaches PV.
73 * CV INT Input The current counter value.
74 *
75 * TP Generate a fixed duration pulse.
76 * IN BOOL Input A rising edge starts the fixed duration pulse.
77 * PT TIME Input The duration of the pulse.
78 * Q BOOL Output Set to true during the pulse.
79 * ET TIME Output The elapsed time of the pulse.
80 *
81 * TON Turn an output on after a delay.
82 * IN BOOL Input A rising edge starts the timer.
83 * PT TIME Input The duration of timer.
84 * Q BOOL Output Set to true after delay PT.
85 * ET TIME Output The elapsed time of the timer.
86 *
87 * TOF Turn an output off after a delay.
88 * IN BOOL Input A rising edge starts the timer.
89 * PT TIME Input The duration of timer.
90 * Q BOOL Output Set to false after delay PT.

112uint8_t iec_sr(iec_t *q1, iec_t s1, iec_t r) {
113 iec_type_allowed(s1, IEC_T_BOOL);
114 iec_type_allowed(r, IEC_T_BOOL);
115
116 iec_totype(q1, IEC_T_BOOL);
117 iec_set_maintain(*q1);
118
119 iec_set_value(*q1, (iec_get_value(s1) || (iec_get_value((*q1)) && !(iec_get_value(r)))));
120
121 return IEC_OK;
122}

153uint8_t iec_r_trig(iec_t *q, iec_t clk) {
154 iec_type_allowed(clk, IEC_T_BOOL);
155
156 iec_totype(q, IEC_T_BOOL);
157 iec_set_maintain(*q);
158
159 iec_set_value(*q, (iec_get_value(clk) && !(iec_is_flag1(*q))));
160 if (iec_get_value(clk))
161 iec_set_flag1(*q);
162 else
163 iec_unset_flag1(*q);
164
165 return IEC_OK;

202uint8_t iec_ctu(iec_t *q, iec_t cu, iec_t r, iec_t pv, iec_t *cv) {
203 iec_type_allowed(cu, IEC_T_R_EDGE);
204 iec_type_allowed(r, IEC_T_BOOL);
205 iec_type_allowed(pv, IEC_T_INT);
206 iec_type_allowed(*cv, IEC_T_INT);
207
208 iec_totype(q, IEC_T_BOOL);
209
210 if (iec_get_value(r)) {
211 iec_set_value(*cv, 0);
212 } else if (iec_get_value(cu) && (iec_get_value(*cv) < iec_get_value(pv)))
213 iec_set_value(*cv, iec_get_value(*cv) + 1);
214
215 iec_set_value(*q, iec_get_value(*cv) >= iec_get_value(pv))
216
217 return IEC_OK;
218}

257uint8_t iec_initialize_timer(iec_t *timer, iec_t pt) {
258 iec_type_allowed(*timer, IEC_T_TIMER);
259 iec_type_allowed(pt, IEC_T_TIME);
260
261 iec_timer(*timer)->q = false;
262 iec_timer(*timer)->pt = iec_get_value(pt);
263 iec_timer(*timer)->et = 0;
264 iec_timer(*timer)->timer_run = false;

282uint8_t iec_tp(iec_t *timer, iec_t in, iec_t pt, iec_t *et) {
283 iec_type_allowed(*timer, IEC_T_TIMER);
284
285 if (!iec_is_initialized(*timer))
286 iec_initialize_timer(timer, pt);
287
288 if (iec_get_value(in)) {
289 if (iec_timer(*timer)->t0 == 0) {
290 iec_timer(*timer)->t0 = hw_millis();
291 }
292 }
293
294 if (iec_timer(*timer)->t0 > 0) {
295 iec_timer(*timer)->et = hw_millis() - iec_timer(*timer)->t0;
296 if (iec_timer(*timer)->et >= iec_timer(*timer)->pt) {
297 iec_timer(*timer)->et = iec_timer(*timer)->pt;
298 }
299 }
300
301 if (!iec_get_value(in) && iec_timer(*timer)->et == iec_timer(*timer)->pt) {
302 iec_timer(*timer)->t0 = 0;
303 iec_timer(*timer)->et = 0;
304 }
305
306 iec_timer(*timer)->q = iec_timer(*timer)->et > 0 && iec_timer(*timer)->et < iec_timer(*timer)->pt;
307
308 if (*et != NULL) {
309 iec_type_allowed(*et, IEC_T_TIME);
310 iec_set_value(*et, iec_timer(*timer)->et);
311 }
312
313 return IEC_OK;
314}
