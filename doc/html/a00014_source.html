277 iec_type_allowed(*timer, IEC_T_TIMER);
278
279 if (!iec_is_initialized(*timer))
280 iec_initialize_timer(timer, pt);
281
282 if (iec_get_value(in)) {

291 iec_timer(*timer)->et = iec_timer(*timer)->pt;
292 }
293 }

296 iec_timer(*timer)->t0 = 0;
297 iec_timer(*timer)->et = 0;
