104uint8_t iec_max(iec_t *result, stack_t *list) {
105 if (*result == NULL || *list == NULL)
106 return IEC_NLL;
107
108 iec_t tmp = IEC_ALLOC;
109 iec_init(&tmp, IEC_T_NULL);
110 iec_move(result, (iec_t) stack_pop(*list));
111 iec_anytype_allowed(*result, ANY_NUM,,,,,);
112
113 while (!stack_empty(*list)) {
114 iec_move(&tmp, (iec_t) stack_pop(*list));
115 iec_anytype_allowed(tmp, ANY_NUM,,,,,);
116 if ((iec_get_value(*result)) < (iec_get_value(tmp))) {
117 iec_move(result, tmp);
118 }
119 }
120
121 iec_deinit(&tmp);
122 return IEC_OK;
123}

164uint8_t iec_limit(iec_t *result, iec_t v, iec_t min, iec_t max) {
165 iec_anytype_allowed(v, ANY_NUM,,,,,);
166 iec_anytype_allowed(min, ANY_NUM,,,,,);
167 iec_anytype_allowed(max, ANY_NUM,,,,,);
168
169 if ((iec_get_value(v)) < (iec_get_value(min)))
170 iec_move(result, min);
171 else if ((iec_get_value(v)) > (iec_get_value(max)))
172 iec_move(result, max);
173 else
174 iec_move(result, v);
175
176 return IEC_OK;
177}

188uint8_t iec_mux(iec_t *result, iec_t v1, stack_t *list) {
189 iec_anytype_allowed(v1, ANY_INT,,,,,);
190
191 if (iec_get_value(v1) <= 0) {
192 iec_move(result, (iec_t) stack_pop(*list));
193 stack_flush(*list);
194 return IEC_OK;
195 }
196
197 maxuint_t cnt = iec_get_value(v1);
198 while (!stack_empty(*list)) {
199 if (cnt-- == 0)
200 break;
201 iec_move(result, (iec_t) stack_pop(*list));
202 }
203
204 stack_flush(*list);
205
206 return IEC_OK;
207}
