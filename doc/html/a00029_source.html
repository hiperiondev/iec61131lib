175// This is the main processing body of the algorithm. It operates
176// on each full 32-bits of input.
177#define DOBLOCK(h1, k1) \
178 do { \
179 k1 *= C1; \
180 k1 = rotl32(k1, 15); \
181 k1 *= C2; \
182 \
183 h1 ^= k1; \
184 h1 = rotl32(h1, 13); \
185 h1 = h1 * 5 + 0xe6546b64; \
186 } while (0)

188// Append unaligned bytes to carry, forcing hash churn if we have 4 bytes
189// cnt=bytes to process, h1=name of h1 var, c=carry, n=bytes in c,
190// ptr/len=payload
191#define DOBYTES(cnt, h1, c, n, ptr, len) \
192 do { \
193 int _i = cnt; \
194 while (_i--) { \
195 c = c >> 8 | *ptr++ << 24; \
196 n++; \
197 len--; \
198 if (n == 4) { \
199 DOBLOCK(h1, c); \
200 n = 0; \
201 } \
202 } \
203 } while (0)

207// Main hashing function. Initialize carry to 0 and h1 to 0 or an initial seed
208// if wanted. Both ph1 and pcarry are required arguments.
209void PMurHash32_Process(uint32_t *ph1, uint32_t *pcarry, const void *key, int len) {
210 uint32_t h1 = *ph1;
211 uint32_t c = *pcarry;
212
213 const uint8_t *ptr = (uint8_t*) key;
214 const uint8_t *end;
215
216 // Extract carry count from low 2 bits of c value
217 int n = c & 3;
218
219#if defined(UNALIGNED_SAFE)
220 // This CPU handles unaligned word access
221
222 // Consume any carry bytes
223 int i = (4 - n) & 3;
224 if (i && i <= len) {
225 DOBYTES(i, h1, c, n, ptr, len);
226 }
227
228 // Process 32-bit chunks
229 end = ptr + len / 4 * 4;
230 for (; ptr < end; ptr += 4) {
231 uint32_t k1 = READ_UINT32(ptr);
232 DOBLOCK(h1, k1);
233 }
234
235 #else //UNALIGNED_SAFE
236 // This CPU does not handle unaligned word access
237
238 // Consume enough so that the next data byte is word aligned
239 int i = -(long) ptr & 3;
240 if (i && i <= len) {
241 DOBYTES(i, h1, c, n, ptr, len);
242 }
243
244 // We're now aligned. Process in aligned blocks. Specialise for each
245 // possible carry count
246 end = ptr + len / 4 * 4;
247 switch (n) { // how many bytes in c
248 case 0: // c=[----] w=[3210] b=[3210]=w c'=[----]
249 for (; ptr < end; ptr += 4) {
250 uint32_t k1 = READ_UINT32(ptr);
251 DOBLOCK(h1, k1);
252 }
253 break;
254 case 1: // c=[0---] w=[4321] b=[3210]=c>>24|w<<8 c'=[4---]
255 for (; ptr < end; ptr += 4) {
256 uint32_t k1 = c >> 24;
257 c = READ_UINT32(ptr);
258 k1 |= c << 8;
259 DOBLOCK(h1, k1);
260 }
261 break;
262 case 2: // c=[10--] w=[5432] b=[3210]=c>>16|w<<16 c'=[54--]
263 for (; ptr < end; ptr += 4) {
264 uint32_t k1 = c >> 16;
265 c = READ_UINT32(ptr);
266 k1 |= c << 16;
267 DOBLOCK(h1, k1);
268 }
269 break;
270 case 3: // c=[210-] w=[6543] b=[3210]=c>>8|w<<24 c'=[654-]
271 for (; ptr < end; ptr += 4) {
272 uint32_t k1 = c >> 8;
273 c = READ_UINT32(ptr);
274 k1 |= c << 24;
275 DOBLOCK(h1, k1);
276 }
277 }
278#endif //UNALIGNED_SAFE
279
280 // Advance over whole 32-bit chunks, possibly leaving 1..3 bytes
281 len -= len / 4 * 4;
282
283 // Append any remaining bytes into carry
284 DOBYTES(len, h1, c, n, ptr, len);
285
286 // Copy out new running hash and carry
287 *ph1 = h1;
288 *pcarry = (c & ~0xff) | n;
289}

293// Finalize a hash. To match the original Murmur3A the total_length must be
294// provided
295uint32_t PMurHash32_Result(uint32_t h, uint32_t carry, uint32_t total_length) {
296 uint32_t k1;
297 int n = carry & 3;
298 if (n) {
299 k1 = carry >> (4 - n) * 8;
300 k1 *= C1;
301 k1 = rotl32(k1, 15);
302 k1 *= C2;
303 h ^= k1;
304 }
305 h ^= total_length;
306
307 // fmix
308 h ^= h >> 16;
309 h *= 0x85ebca6b;
310 h ^= h >> 13;
311 h *= 0xc2b2ae35;
312 h ^= h >> 16;
313
314 return h;
315}

320uint32_t PMurHash32(uint32_t seed, const void *key, int len) {
321 uint32_t h1 = seed, carry = 0;
322 PMurHash32_Process(&h1, &carry, key, len);
323 return PMurHash32_Result(h1, carry, len);
