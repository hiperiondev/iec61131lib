146uint8_t iec_identify_literal(str_t *str, uint8_t *iectype) {
147 uint8_t n;
148
149 trimAll(str);
150 toUpperCase(str);
151
152 // replace boolean type
153 replaceAllOccurrences(str, "TRUE", "1");
154 replaceAllOccurrences(str, "FALSE", "0");
155
156 // delete _
157 str_t *tmp = EMPTY_STRING(255);
158 while(indexOfChar(str, '_', 0) != -1) {
159 substringAfter(str, tmp, "_");
160 substringBefore(str, str, "_");
161 concatString(str, tmp);
162 }
163
164 // search typed literal
165 *iectype = IEC_LIT_NONE;
166 for (n = 0; n < 32; n++) {
167 if (isStringStartsWith(str, IEC_IECTYPE_PFX[n], 0)) {
168 *iectype = n;
169 break;
170 }
171 }
172 if(*iectype != IEC_LIT_NONE)
173 substringAfter(str, str, "#");
174
175 if (*iectype < IEC_LIT_INTEGER)
176 return IEC_LIT_NONE;
177
178 // search data prefix
179 uint8_t datatype = IEC_LIT_NONE;
180 for (n = 0; n < 10; n++) {
181 if (isStringStartsWith(str, IEC_DATAFORMAT_PFX[n], 0)) {
182 datatype = n;
183 break;
184 }
185 }
186 if (datatype != IEC_LIT_NONE) {
187 substringAfter(str, str, "#");
188 return IEC_LITERAL_FORMAT[datatype];
189 }
190
191 if(isBuffStringEquals(str, NEW_STRING(2, "0")) || isBuffStringEquals(str, NEW_STRING(2, "1")))
192 return IEC_LIT_BOOLEAN;
193
194
195 if ((indexOfChar(str, 'e', 0) != -1) || (indexOfChar(str, 'E', 0) != -1))
196 return IEC_LIT_REAL_EXP;
197
198 if (indexOfChar(str, '.', 0) != -1)
199 return IEC_LIT_REAL;
200
201 return IEC_LIT_INTEGER;
202}
