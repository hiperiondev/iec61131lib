218uint8_t iec_string_set(iec_t *result, char *str, bool wstr, bool hash) {
219 if (wstr) {
220 iec_totype(result, IEC_T_WSTRING);
221 } else {
222 iec_totype(result, IEC_T_STRING);
223 }
224
225 (*result)->value = malloc(sizeof(string_t));
226 ((string_t*) ((*result)->value))->str = newStringWithLength2(&(str_t) { 0 }, str, strlen(str));
227
228 if (hash) {
229 ((string_t*) ((*result)->value))->hash = PMurHash32(STR_SEED_HASH, stringValue(((string_t*) ((*result)->value))->str),
230 stringLength(((string_t*) ((*result)->value))->str));
231 }
232
233 return IEC_OK;
234}
