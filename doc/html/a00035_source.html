67uint8_t iec_add(iec_t *result, stack_t *list) {
68 if (*result == NULL || list == NULL)
69 return IEC_NLL;
70
71 iec_t tmp = IEC_ALLOC;
72 iec_init(&tmp, IEC_T_NULL);
73 iec_move(result, (iec_t) stack_pop(*list));
74 iec_anytype_allowed(*result, ANY_MAGNITUDE,,,,,);
75
76 while (!stack_empty(*list)) {
77 iec_move(&tmp, (iec_t) stack_pop(*list));
78 iec_anytype_allowed(tmp, ANY_MAGNITUDE,,,,,);
79 iec_type_promote(result, tmp->type);
80 iec_set_value(*result, (iec_get_value(*result)) + (iec_get_value(tmp)));
81 }
82
83 iec_deinit(&tmp);
84 return IEC_OK;
85}

144uint8_t iec_div(iec_t *result, iec_t v1, iec_t v2) {
145 iec_anytype_allowed(v1, ANY_MAGNITUDE,ANY_NUM,,,,);
146 iec_anytype_allowed(v2, ANY_MAGNITUDE,ANY_NUM,,,,);

172uint8_t iec_mod(iec_t *result, iec_t v1, iec_t v2) {
173 iec_anytype_allowed(v1, ANY_INT,,,,,);
174 iec_anytype_allowed(v2, ANY_INT,,,,,);
