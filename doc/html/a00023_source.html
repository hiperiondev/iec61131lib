41/*
42 * Summary:
43 *
44 * Function Parameter Type Parameters Description
45 * SHL 1:ANY_INTEGRAL;2:ANY_NUM 2 Shifts an integer value to the left.
46 * SHR 1:ANY_INTEGRAL;2:ANY_NUM 2 Shifts an integer value to the right.
47 * ROL 1:ANY_INTEGRAL;2:ANY_NUM 2 Rotates an integer value to the left.
48 * ROR 1:ANY_INTEGRAL;2:ANY_NUM 2 Rotates an integer value to the right
49 */

51#define U8V(v) ((uint8_t)(v) & 0xFFU)
52#define U16V(v) ((uint16_t)(v) & 0xFFFFU)
53#define U32V(v) ((uint32_t)(v) & 0xFFFFFFFFU)
54#define U64V(v) ((uint64_t)(v) & 0xFFFFFFFFFFFFFFFFU)
55
56#define rotl8(v, n) (U8V((uint8_t)(v) << (n)) | ((uint8_t)(v) >> (8 - (n))))
57#define rotl16(v, n) (U16V((uint16_t)(v) << (n)) | ((uint16_t)(v) >> (16 - (n))))
58#define rotl32(v, n) (U32V((uint32_t)(v) << (n)) | ((uint32_t)(v) >> (32 - (n))))
59#define rotl64(v, n) (U64V((uint64_t)(v) << (n)) | ((uint64_t)(v) >> (64 - (n))))
60
61#define rotr8(v, n) rotl8(v, 8 - (n))
62#define rotr16(v, n) rotl16(v, 16 - (n))
63#define rotr32(v, n) rotl32(v, 32 - (n))
64#define rotr64(v, n) rotl64(v, 64 - (n))

75uint8_t iec_shl(iec_t *result, iec_t v1, iec_t v2) {
76 iec_anytype_allowed(v1, ANY_INTEGRAL,,,,,);
77 iec_anytype_allowed(v2, ANY_NUM,,,,,);
78 iec_type_promote(result, v1->type);
79
80 maxuint_t _v1 = iec_get_value(v1);
81 maxuint_t _v2 = iec_get_value(v2);
82
83 if (sign(iec_get_value(v2))) {
84 _v1 = _v1 << _v2;
85 } else {
86 _v1 = _v1 >> _v2;
87 }
88
89 iec_set_value(*result, _v1);
90
91 return IEC_OK;
92}

129uint8_t iec_rol(iec_t *result, iec_t v1, iec_t v2) {
130 iec_anytype_allowed(v1, ANY_INTEGRAL,,,,,);
131 iec_anytype_allowed(v2, ANY_NUM,,,,,);
132 iec_type_promote(result, v1->type);
133
134 maxuint_t _v1 = iec_get_value(v1);
135 maxuint_t _v2 = iec_get_value(v2);
136 if (sign(iec_get_value(v2))) {
137 switch (IEC_T_SIZEOF[v1->type]) {
138 case 8:
139 _v1 = rotl8(_v1, _v2);

141 case 16:

143 break;
144 case 32:

146 break;
147 case 64:

152 }
